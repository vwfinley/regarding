\section{Problem}
%Refer to Figure \ref{fig:fig_a1}.
\paragraph{Given:}
\begin{itemize}
\item Some semi-circle of radius $r$,
\item \emph{and} a vertical line, of length $c$, inscribed on the semi-circle.
\end{itemize}

\paragraph{Find:}
\begin{enumerate}
\item A procedure to construct tangent ogives.
\item Relationships among tangent ogive elements.
\end{enumerate}
