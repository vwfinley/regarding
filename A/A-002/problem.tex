\section{Problem}

\paragraph{Given:}
\begin{itemize}
  \item As shown in Figure \ref{fig:fig_a}:
  \item Two lines, $L_1$ and $L_2$, that intersect at an angle $\theta$, at some point $I$,
  \item \emph{and} a fillet $C$, of radius $r$, that is tangent to both $L_1$ and $L_2$;
\end{itemize}

\paragraph{Find:}
\begin{enumerate}
  \item The location of the center of $C$ relative to $I$,
  \item \emph{and} the locations where the $C$ is tangent to $L_1$ and $L_2$,
  \item \emph{and} the location on $C$ closest to $I$.
\end{enumerate}
