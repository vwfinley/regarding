\section{Problem}
Refer to Figure \ref{fig:fig_a}.
\paragraph{Given:}
\begin{itemize}
  \item Two coplanar circles $A$ and $B$,
  \item \emph{with} center points $C_A$ and $C_B$ respectively,
  \item \emph{and} having radii $r_A$ and $r_B$ respectively,
  \item \emph{where} $r_A < r_B$;
\end{itemize}

\begin{figure}[h]
\centering
\input{fig_a.tex}
\caption{Coplanar circles $A$ and $B$}
\label{fig:fig_a}
\end{figure}

\paragraph{Find:}
\begin{enumerate}
  \item The critical location of $C_A$ where $A$ and $B$ are concentric.
  \item The critical location of $C_A$ where $C_B$ lies on $A$.
  \item The critical locations of $C_A$ where $A$ and $B$ are tangent to each other.
  \item The critical location of $C_A$ where $C_A$ lies on $B$.
  \item All locations of $C_A$ where $A$ otherwise overlaps $B$.
  \item All locations of $C_A$ where $A$ has no overlap with $B$.
\end{enumerate}
