\section{Example}

\paragraph{Given:}
\begin{itemize}
\item Two circles $A$ and $B$ whose radii are $r_A = 1.0$ and $r_B = 3.0$ units respectively,
\item \emph{and} circle centers $C_A$ and $C_B$ separated by a distance of 2.5 units.
\end{itemize}

\paragraph{Find:}
\begin{itemize}
\item Which solution case best matches the given arrangement?
\end{itemize}

\paragraph{Steps:}
\begin{enumerate}
\item
Let: 
\begin{align*}
	x_A &= 2.5 \\
	r_A &= 1.0 \\
	r_B &= 3.0
\end{align*}

\item
From the solution section, use the inequality for case 5:
\begin{equation*}
	r_B - r_A < x_A < r_B \\
\end{equation*}

\item
Plug in values for $r_A$ and $r_B$,
\begin{equation*}
	3.0 - 1.0 < x_A < 3.0
\end{equation*}
\begin{equation*}
	2.0 < x_A < 3.0
\end{equation*}

\item
Since $x_A = 2.5$ is between 2.0 and 3.0, $x_A$ satisfies the inequality for case 5.  Therefore the relationship
between circles $A$ and $B$ are classified by case 5.

\end{enumerate}
