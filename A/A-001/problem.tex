\section{Problem}

\paragraph{Given:}

\begin{itemize}
  \item Circle $C$ of radius $r$,
  \item \emph{and} line $L$ sloped at an angle $\alpha$ with respect to the horizontal,
  \item \emph{and} point $P$ where $L$ is tangent to $C$;
\end{itemize}

\paragraph{Find:}

The location of $P$ relative to the center of $C$.
