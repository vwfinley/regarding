\subsection{From inside devcontainers}
- VSCode
- .devcontainer/devcontainer.json
- Dockerfile
- Additional packages

\begin{mdframed}[backgroundcolor=black!20,leftmargin=0.0cm,skipabove=0.4cm,hidealllines=true,
  innerleftmargin=0.1cm,innerrightmargin=0.1cm,innertopmargin=0.15cm,innerbottommargin=-0.1cm]
\begin{code}
#!/bin/bash

# Run this inside developer container.
INFILE=main
OUTDIR=output
OUTFILE=C-003

mkdir $OUTDIR

latex -output-format=pdf \
  -output-directory=$OUTDIR \
  $INFILE.tex

pushd $OUTDIR
mv $INFILE.pdf $OUTFILE.pdf
popd
\end{code}
\end{mdframed}


\subsection{From host (outside guest container)}

\begin{mdframed}[backgroundcolor=black!20,leftmargin=0.0cm,skipabove=0.4cm,hidealllines=true,
  innerleftmargin=0.1cm,innerrightmargin=0.1cm,innertopmargin=0.15cm,innerbottommargin=-0.1cm]
\begin{code}
#!/bin/bash

# Run this from host.
INFILE=main
OUTDIR=output
OUTFILE=C-003

mkdir $OUTDIR

docker run --rm -it -v $(pwd):/workspace \
  ghcr.io/vwfinley/texlive-dev:latest \
  latex -output-format=pdf \
  -output-directory=$OUTDIR $INFILE.tex
  
pushd $OUTDIR
mv $INFILE.pdf $OUTFILE.pdf
popd
\end{code}
\end{mdframed}
