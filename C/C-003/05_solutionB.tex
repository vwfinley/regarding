\section{Solution B: Roll own container from base image}
- Find base devcontainer image.
- Add missing needed packages
- Fix warnings/errors (mktexpk for generating missing fonts)
- Integrate/Use container in project.

Add TexLive tlmgr packages in Docker container and publish.
vs. add packages 




\begin{mdframed}[backgroundcolor=black!20,leftmargin=0.0cm,skipabove=0.4cm,hidealllines=true,
  innerleftmargin=0.1cm,innerrightmargin=0.1cm,innertopmargin=0.15cm,innerbottommargin=-0.1cm]
\begin{code}
ARG IMG=mcr.microsoft.com/devcontainers/base
ARG TAG=dev-debian-12

FROM mcr.microsoft.com/devcontainers/base:dev-debian-12

LABEL org.opencontainers.image.source https://github.com/vwfinley/texlive

ENV TEXLIVE_VERSION=2025
ENV TEXLIVE_SCHEME=small
ENV PATH="$PATH:/usr/local/texlive/${TEXLIVE_VERSION}/bin/x86_64-linux"


RUN apt-get update

WORKDIR /tmp 

RUN <<EOF
    wget https://mirror.ctan.org/systems/texlive/tlnet/install-tl-unx.tar.gz
    zcat < install-tl-unx.tar.gz | tar xf -
    cd install-tl-*
    perl ./install-tl --scheme=${TEXLIVE_SCHEME} \
      --no-interaction --no-doc-install --no-src-install
    tlmgr update --all
    cd ..
    rm -r install*
EOF

WORKDIR /workspace 

USER vscode


\end{code}
\end{mdframed}
