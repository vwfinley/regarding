\section{Problem}
\paragraph{Given:}
\begin{itemize}
\item A positively sloped line $L_s$ passing through the origin,
\item \emph{and} a horizontal line $L_h$ passing some distance $h$ above the origin.
\end{itemize}
\paragraph{Find:}
\begin{enumerate}
\item The circle that:
\begin{itemize}
\item passes through the origin,
\item \emph{and} is tangent to horizontal line $L_h$,
\item \emph{and} has its center somewhere on sloped line $L_s$.
\end{itemize}
\item A geometric construction for fitting and placing such a circle.
\item A trigonometric method for fitting and placing such a circle.
\item Locations of the circle center and tangent point.
\item The radius of the circle.
\end{enumerate}
