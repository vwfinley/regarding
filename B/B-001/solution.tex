\subsection{Locate $p_g$}

\begin{equation}
	y_g = y_s - P
\end{equation}

\begin{equation}
	x_g = \sqrt{(R1)^2 - y_g^2}
\end{equation}

\begin{equation}
	\theta_g = \arcsin \left( \frac{y_g}{R1} \right)
\end{equation}


\subsection{Locate $p_2$}
\begin{equation}
	x_2 = R2 \cdot \cos(\theta_g)
\end{equation}

\begin{equation}
	y_2 = R2 \cdot \sin(\theta_g)
\end{equation}


\subsection{Locate $p_d$}
\begin{equation}
	x_d = x_2 - T/2
\end{equation}

\begin{equation}
	y_d = \sqrt{(R2)^2 - x_d^2}
\end{equation}


\subsection{Locate $p_3$}
\begin{equation}
	x_3 = 2 \cdot x_d
\end{equation}

\begin{equation}
	y_3 = 0
\end{equation}

\end{equation}

\end{equation}

\begin{equation}
	p_s = (x_s - x_g, y_s - y_g)

	p_2 = (x_2, y_2)
\end{equation}

	p_d = (x_2 - x_d, y_2 - y_d)
